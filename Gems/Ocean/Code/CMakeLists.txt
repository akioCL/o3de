
# Currently we are in the Code folder: ${CMAKE_CURRENT_LIST_DIR}
# Get the platform specific folder ${pal_dir} for the current folder: ${CMAKE_CURRENT_LIST_DIR}/Platform/${PAL_PLATFORM_NAME}
# Note: o3de_pal_dir will take care of the details for us, as this may be a restricted platform
#       in which case it will see if that platform is present here or in the restricted folder.
#       i.e. It could here in our gem : Gems/Ocean/Code/Platform/<platorm_name>  or
#            <restricted_folder>/<platform_name>/Gems/Ocean/Code
o3de_pal_dir(pal_dir ${CMAKE_CURRENT_LIST_DIR}/Platform/${PAL_PLATFORM_NAME} "${gem_restricted_path}" "${gem_path}" "${gem_name}")

# Now that we have the platform abstraction layer (PAL) folder for this folder, thats where we will find the
# traits for this platform. Traits for a platform are defines for things like whether or not something in this gem
# is supported by this platform.
include(${pal_dir}/PAL_${PAL_PLATFORM_NAME_LOWERCASE}.cmake)

# Add the Ocean.Static target
# Note: We include the common files and the platform specific files which are set in ocean_common_files.cmake
# and in ${pal_dir}/ocean_${PAL_PLATFORM_NAME_LOWERCASE}_files.cmake
ly_add_target(
    NAME Ocean.Static STATIC
    NAMESPACE Gem
    FILES_CMAKE
        ocean_files.cmake
        ${pal_dir}/ocean_${PAL_PLATFORM_NAME_LOWERCASE}_files.cmake
    INCLUDE_DIRECTORIES
        PUBLIC
            Include
        PRIVATE
            Source
    BUILD_DEPENDENCIES
        PUBLIC
            AZ::AzCore
            AZ::AzFramework
            Gem::Atom_RPI.Public
            Gem::Atom_Utils.Static
            Gem::Atom_Feature_Common
            Gem::AtomLyIntegration_CommonFeatures.Static
            Gem::LmbrCentral
)

# Here add Ocean target, it depends on the Ocean.Static
ly_add_target(
    NAME Ocean ${PAL_TRAIT_MONOLITHIC_DRIVEN_MODULE_TYPE}
    NAMESPACE Gem
    FILES_CMAKE
        ocean_shared_files.cmake
        ${pal_dir}/ocean_shared_${PAL_PLATFORM_NAME_LOWERCASE}_files.cmake
    INCLUDE_DIRECTORIES
        PUBLIC
            Include
        PRIVATE
            Source
    BUILD_DEPENDENCIES
        PRIVATE
            Gem::Ocean.Static
            Gem::LmbrCentral
)

# By default, we will specify that the above target Ocean would be used by
# Client and Server type targets when this gem is enabled.  If you don't want it
# active in Clients or Servers by default, delete one of both of the following lines:
ly_create_alias(NAME Ocean.Clients NAMESPACE Gem TARGETS Gem::Ocean)
ly_create_alias(NAME Ocean.Servers NAMESPACE Gem TARGETS Gem::Ocean)

# If we are on a host platform, we want to add the host tools targets like the Ocean.Editor target which
# will also depend on Ocean.Static
if(PAL_TRAIT_BUILD_HOST_TOOLS)
    ly_add_target(
        NAME Ocean.Editor.Static STATIC
        NAMESPACE Gem
        FILES_CMAKE
            ocean_editor_files.cmake
        INCLUDE_DIRECTORIES
            PRIVATE
                Source
            PUBLIC
                Include
        BUILD_DEPENDENCIES
            PUBLIC
                AZ::AzToolsFramework
                Gem::LmbrCentral
                Gem::Ocean.Static
    )

    ly_add_target(
        NAME Ocean.Editor GEM_MODULE
        NAMESPACE Gem
        AUTOMOC
        FILES_CMAKE
            ocean_editor_shared_files.cmake
        INCLUDE_DIRECTORIES
            PRIVATE
                Source
            PUBLIC
                Include
        BUILD_DEPENDENCIES
            PUBLIC
                Gem::Ocean.Editor.Static
                Gem::LmbrCentral
    )

    # By default, we will specify that the above target Ocean would be used by
    # Tool and Builder type targets when this gem is enabled.  If you don't want it
    # active in Tools or Builders by default, delete one of both of the following lines:
    ly_create_alias(NAME Ocean.Tools    NAMESPACE Gem TARGETS Gem::Ocean.Editor)
    ly_create_alias(NAME Ocean.Builders NAMESPACE Gem TARGETS Gem::Ocean.Editor)


endif()

################################################################################
# Tests
################################################################################
# See if globally, tests are supported
if(PAL_TRAIT_BUILD_TESTS_SUPPORTED)
    # We globally support tests, see if we support tests on this platform for Ocean.Static
    if(PAL_TRAIT_OCEAN_TEST_SUPPORTED)
        # We support Ocean.Tests on this platform, add Ocean.Tests target which depends on Ocean.Static
        ly_add_target(
            NAME Ocean.Tests ${PAL_TRAIT_TEST_TARGET_TYPE}
            NAMESPACE Gem
            FILES_CMAKE
                ocean_files.cmake
                ocean_tests_files.cmake
            INCLUDE_DIRECTORIES
                PRIVATE
                    Tests
                    Source
            BUILD_DEPENDENCIES
                PRIVATE
                    AZ::AzTest
                    AZ::AzFramework
                    Gem::Ocean.Static
        )

        # Add Ocean.Tests to googletest
        ly_add_googletest(
            NAME Gem::Ocean.Tests
        )
    endif()

    # If we are a host platform we want to add tools test like editor tests here
    if(PAL_TRAIT_BUILD_HOST_TOOLS)
        # We are a host platform, see if Editor tests are supported on this platform
        if(PAL_TRAIT_OCEAN_EDITOR_TEST_SUPPORTED)
            # We support Ocean.Editor.Tests on this platform, add Ocean.Editor.Tests target which depends on Ocean.Editor
            ly_add_target(
                NAME Ocean.Editor.Tests ${PAL_TRAIT_TEST_TARGET_TYPE}
                NAMESPACE Gem
                FILES_CMAKE
                    ocean_editor_tests_files.cmake
                INCLUDE_DIRECTORIES
                    PRIVATE
                        Tests
                        Source
                BUILD_DEPENDENCIES
                    PRIVATE
                        AZ::AzTest
                        Gem::Ocean.Editor
            )

            # Add Ocean.Editor.Tests to googletest
            ly_add_googletest(
                NAME Gem::Ocean.Editor.Tests
            )
        endif()
    endif()
endif()
